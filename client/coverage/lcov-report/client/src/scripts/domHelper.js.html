export const delay = (callback, ms) => {
  let timer = 0;
  return () => {
    const context = this;
    /*eslint-disable */
    const args = arguments;
    /* eslint-enable */
    clearTimeout(timer);
    timer = setTimeout(() => {
      callback.apply(context, args);
    }, ms || 0);
  };
};
